### 1.1 Desktop: expanded, но с `<Reveal>` stagger + тонкий hover

```jsx
function LegalCard({ item, index, isDesktop }) {
  const [expanded, setExpanded] = useState(isDesktop);
  useEffect(() => setExpanded(isDesktop), [isDesktop]);
  
  return (
    <Reveal delay={index*80}>
      <article style={{
        background:'#15181C', border:`1px solid ${item.color}33`, borderRadius:12,
        padding:20, transition:'border-color 0.2s'
      }}
      onMouseEnter={e => e.currentTarget.style.borderColor = `${item.color}99`}
      onMouseLeave={e => e.currentTarget.style.borderColor = `${item.color}33`}>
        <button
          onClick={() => !isDesktop && setExpanded(x => !x)}
          style={{
            background:'none', border:'none', width:'100%', display:'flex', alignItems:'center',
            gap:12, textAlign:'left', cursor: isDesktop ? 'default' : 'pointer'
          }}
          aria-expanded={expanded}
          disabled={isDesktop}
        >
          <item.icon size={20} color={item.color}/>
          <h3 style={{flex:1, fontSize:16, color:'#EAEAEA', margin:0}}>{item.title}</h3>
          {!isDesktop && <ChevronDown style={{transform: expanded ? 'rotate(180deg)' : '', transition:'transform 0.2s'}}/>}
        </button>
        <div style={{maxHeight: expanded ? 500 : 0, overflow:'hidden', transition:'max-height 0.3s'}}>
          <p style={{color:'#8E8E93', fontSize:14, lineHeight:1.6, marginTop:12}}>{item.text}</p>
        </div>
      </article>
    </Reveal>
  );
}
```

### 1.2 Mobile detect

```jsx
function useIsDesktop() {
  const [desktop, setDesktop] = useState(true);
  useEffect(() => {
    const mq = window.matchMedia('(min-width: 768px)');
    const handler = () => setDesktop(mq.matches);
    handler();
    mq.addEventListener('change', handler);
    return () => mq.removeEventListener('change', handler);
  }, []);
  return desktop;
}
```

```jsx
function NDAGate() {
  const [agreed, setAgreed] = useState(false);
  return (
    <Reveal delay={600}>
      <div style={{marginTop:40, padding:24, background:'#15181C', border:'1px solid #F4A26133', borderRadius:12}}>

            Я прочитал и согласен с указанными оговорками. Подтверждаю статус <Tooltip explanation="Квалифицированный инвестор — физ/юр лицо по ФЗ-156, имеющее право инвестировать в closed-end fund'ы">квалифицированного инвестора</Tooltip>.
          </span>
        </label>
        <button
          disabled={!agreed}
          onClick={() => alert('Запрос NDA: отправьте письмо на ir@trendstudio.ru')}
          style={{
            marginTop:20, width:'100%', padding:14,
            background: agreed ? '#F4A261' : '#2A2D31',
            color: agreed ? '#0B0D10' : '#8E8E93',
            border:'none', borderRadius:10, fontSize:15, fontWeight:600,
            cursor: agreed ? 'pointer' : 'not-allowed',
            opacity: agreed ? 1 : 0.5, transition:'all 0.2s'
          }}
        >
          Запросить NDA и доступ к полному PPM →
        </button>
      </div>
    </Reveal>
  );
}
```
