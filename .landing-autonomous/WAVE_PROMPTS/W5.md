## Типовой код

### Waterfall scroll-pinned

```jsx
function WaterfallScrollPinned() {
  const containerRef = useRef(null);
  const [multiplier, setMultiplier] = useState(1.5);
  
  useEffect(() => {
    const handler = () => {
      if (!containerRef.current) return;
      const rect = containerRef.current.getBoundingClientRect();
      const winH = window.innerHeight;
      const progress = Math.max(0, Math.min(1, -rect.top / (rect.height - winH)));
      setMultiplier(0.5 + progress * 4.5);  // 0.5x → 5x
    };
    window.addEventListener('scroll', handler, { passive: true });
    return () => window.removeEventListener('scroll', handler);
  }, []);
  
  return (
    <div ref={containerRef} style={{height:'200vh'}}>
      <div style={{position:'sticky', top:0, height:'100vh'}}>
        <WaterfallBarsInteractive multiplier={multiplier}/>
        <canvas ref={canvasRef}/>  {/* particles */}
      </div>
    </div>
  );
}
```
