## Термин Sheet accordion код

```jsx
const [openRow, setOpenRow] = useState(null);

<table>
  {TERM_ROWS.map((row, i) => (
    <tr key={row.id}>
      <td colSpan={2}>
        <button
          onClick={() => setOpenRow(openRow === row.id ? null : row.id)}
          aria-expanded={openRow === row.id}
          style={{width:'100%', textAlign:'left'}}
        >
          {row.label}
        </button>
        {openRow === row.id && (
          <div style={{padding:16, background:'rgba(244,162,97,0.06)', borderRadius:8}}>
            <strong>{row.value}</strong>
            <p>{row.explanation}</p>
            <em>Impact on LP: {row.impact}</em>
          </div>
        )}
      </td>
    </tr>
  ))}
</table>
```

## Legal flip-card код

```jsx
const [flipped, setFlipped] = useState({});

<div style={{perspective: 1000}}>
  {LEGAL_CARDS.map(c => (
    <div key={c.id} 
      onClick={() => setFlipped({...flipped, [c.id]: !flipped[c.id]})}
      aria-expanded={!!flipped[c.id]}
      style={{
        position:'relative', width:'100%', height:200,
        transformStyle:'preserve-3d',
        transform: flipped[c.id] ? 'rotateY(180deg)' : 'rotateY(0)',
        transition:'transform 600ms cubic-bezier(0.22, 1, 0.36, 1)',
        cursor:'pointer'
      }}
    >
      <div style={{position:'absolute', inset:0, backfaceVisibility:'hidden'}}>
        {/* front: icon + title + teaser */}
      </div>
      <div style={{position:'absolute', inset:0, transform:'rotateY(180deg)', backfaceVisibility:'hidden'}}>
        {/* back: full text */}
      </div>
    </div>
  ))}
</div>
```
