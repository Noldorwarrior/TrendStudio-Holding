### Waterfall cascade с canvas

```jsx
// Canvas для particles
useEffect(() => {
  const canvas = canvasRef.current;
  const ctx = canvas.getContext('2d');
  let particles = Array(50).fill().map(() => ({x:Math.random()*W, y:Math.random()*H, v:1}));
  let raf;
  const animate = () => {
    ctx.fillStyle='rgba(11,13,16,0.1)'; ctx.fillRect(0,0,W,H);
    particles.forEach(p => {
      p.y += p.v;
      if (p.y > H) { p.y = 0; p.x = Math.random()*W; }
      ctx.fillStyle='#F4A261';
      ctx.beginPath(); ctx.arc(p.x, p.y, 2, 0, 2*Math.PI); ctx.fill();
    });
    raf = requestAnimationFrame(animate);
  };
  animate();
  return () => cancelAnimationFrame(raf);
}, []);
```
